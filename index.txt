          const suffix = num.textContent.replace(/[0-9,]/g,'');
          animateCounter(num, parseInt(raw), suffix);
        });
        impactObs.unobserve(e.target);
      }
    });
  }, { threshold: .2 });
  const impactSection = document.getElementById('impact');
  if (impactSection) impactObs.observe(impactSection);
